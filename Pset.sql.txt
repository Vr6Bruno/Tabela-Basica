-- Cria a Função para Calcular Desconto
CREATE FUNCTION lojas.calcular_desconto (
    @preco DECIMAL(10, 2),
    @desconto DECIMAL(5, 2)
)
RETURNS DECIMAL(10, 2)
AS
BEGIN
    DECLARE @preco_com_desconto DECIMAL(10, 2);
    SET @preco_com_desconto = @preco - (@preco * @desconto / 100);
    RETURN @preco_com_desconto;
END;
GO

-- Cria uma Função Escalar para Calcular Total do Pedido
CREATE FUNCTION lojas.calcular_total_pedido (
    @pedido_id DECIMAL(38, 0)
)
RETURNS DECIMAL(10, 2)
AS
BEGIN
    DECLARE @total_pedido DECIMAL(10, 2);
    SELECT @total_pedido = SUM(quantidade * preco_unitario)
    FROM lojas.itens_pedido
    WHERE pedido_id = @pedido_id;
    RETURN @total_pedido;
END;
GO

GO

-- Atualiza o Total do Pedido
UPDATE lojas.pedidos
SET total_pedido = lojas.calcular_total_pedido(pedido_id);
GO

-- Cria uma Função Escalar para Calcular Desconto
CREATE FUNCTION lojas.calcular_desconto (
    @preco_unitario DECIMAL(10, 2),
    @percentual_desconto INT
)
RETURNS DECIMAL(10, 2)
AS
BEGIN
    DECLARE @preco_com_desconto DECIMAL(10, 2);
    SET @preco_com_desconto = @preco_unitario - (@preco_unitario * (@percentual_desconto / 100.0));
    RETURN @preco_com_desconto;
END;

CREATE VIEW lojas.pedidos_detalhados
AS
SELECT
    p.pedido_id,
    p.data_pedido,
    i.item_id,
    i.produto_id,
    i.quantidade,
    i.preco_unitario,
    lojas.calcular_desconto(i.preco_unitario, 10) AS preco_unitario_com_desconto,
    (i.quantidade * i.preco_unitario) AS subtotal
FROM
    lojas.pedidos p
    JOIN lojas.itens_pedido i ON p.pedido_id = i.pedido_id;
GO
